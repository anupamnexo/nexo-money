================================================================================
DATABASE SCHEMA
================================================================================

TABLE 1: companies
  - id (PK), name, gstin, pan, industry, city, plan, created_at

TABLE 2: users
  - id (PK), email (UNIQUE), password_hash, name, role, company_id (FK), created_at
  - Roles: admin, employee, approver

TABLE 3: cards
  - id (PK), company_id (FK), user_id (FK), card_number (masked)
  - card_type, status, daily_limit, monthly_limit, spent_today, spent_month
  - merchant_category_restriction, created_at

TABLE 4: expenses
  - id (PK), company_id (FK), user_id (FK), card_id (FK)
  - amount, currency (INR), merchant_name, category, description, receipt_url
  - gstin_vendor, gst_amount, igst, cgst, sgst
  - status (pending/approved/rejected/auto_approved), approved_by (FK), created_at

TABLE 5: approvals
  - id (PK), expense_id (FK), approver_id (FK), status, comment, created_at

TABLE 6: budgets
  - id (PK), company_id (FK), team, category, monthly_limit, spent_this_month, period, created_at

TABLE 7: vendors
  - id (PK), company_id (FK), name, gstin, total_spent, transaction_count, created_at

TABLE 8: waitlist
  - id (PK), email (UNIQUE), company_name, phone, created_at
